  % containing box, so we check that their length (at the default font
  % size) is not wider than the box.  If not, then keep it as-is.  If
  % it is, then resize it down to fit.
  {
    \def\fnamef{{\fontsize{36}{40}\textbf{\fname}}} % size of fname here
    \newdimen{\namewidth}
    \setlength{\namewidth}{\widthof{\fnamef}}
    \ifthenelse{\lengthtest{\namewidth < \textwidth}}% check if it is too long
    {\fnamef}% keep as-is
    {\resizebox{\textwidth}{!}{\fnamef}}% resize
  }

  {
    \def\lnamef{\Large{\lname}} % change size of lname here
    \newdimen{\namewidth}
    \setlength{\namewidth}{\widthof{\lnamef}}
    \ifthenelse{\lengthtest{\namewidth < \textwidth}}{\lnamef}{\resizebox{\textwidth}{!}{\lnamef}}
  }
